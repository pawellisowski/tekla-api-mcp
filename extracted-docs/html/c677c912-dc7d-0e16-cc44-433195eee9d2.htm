View Class

The View class contains methods related to views. Views are views that contain other drawing objects, excluding other views. Views also contain drawing representations of model objects.

Examples
The following example creates and inserts two new views to the active drawing. A drawing must be open when executing the code.
Copy
using Tekla.Structures.Drawing;
using Tekla.Structures.Geometry3d;

public class Example
{
       public void Example1()
       {
           Beam beam = new Beam(new Point(0, 0, 0), new Point(500, 0, 0));
           beam.Profile.ProfileString = "HEA300";
           beam.Insert();
           Beam column = new Beam(new Point(0, 0, 0), new Point(0, 0, 500));
           column.Profile.ProfileString = "HEA200";
           column.Insert();
           new Model().CommitChanges();

           DrawingHandler myDrawingHandler = new DrawingHandler();
           if(myDrawingHandler.GetConnectionStatus())
           {
               Drawing currentDrawing = myDrawingHandler.GetActiveDrawing();

               CoordinateSystem coordSysPlan = new CoordinateSystem(
                    new Point(), new Vector(1, 0, 0), new Vector(0, 1, 0));

               CoordinateSystem coordSys3d = new CoordinateSystem(
                   new Point(), new Vector(0.5, 0.866, 0), new Vector(-0.296, 0.171, 0.94));

               View newPlanView = new View(Drawing.GetSheet(), coordSysPlan, coordSysPlan,
                   new AABB(new Point(-1000, -1000, -1000), new Point(1000, 1000, 1000)));
               newPlanView.Insert();

               View new3dView = new View(Drawing.GetSheet(), coordSysPlan, coordSys3d,
                   new AABB(new Point(-1000, -1000, -1000), new Point(1000, 1000, 1000)));
               new3dView.Insert();

               currentDrawing.PlaceViews();
           }
       }
}
