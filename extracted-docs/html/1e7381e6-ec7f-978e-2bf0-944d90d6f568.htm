Examples
This example shows how to copy the Distance instances in a DistanceList to an array.
Copy
using Tekla.Structures.Datatype;

public class Example
{
       public void Example1()
       {
           // We'll use this distance list to demonstrate the CopyTo method.
           DistanceList distanceList = new DistanceList(new Distance[]
                                                       {
                                                           new Distance(30.2),
                                                           new Distance(30.2),
                                                           new Distance(30.2),
                                                           new Distance(50),
                                                           new Distance(2.3),
                                                           new Distance(2.3),
                                                       }); 

           Distance[] array = new Distance[distanceList.Count];
           distanceList.CopyTo(array, 0);
       }
}
