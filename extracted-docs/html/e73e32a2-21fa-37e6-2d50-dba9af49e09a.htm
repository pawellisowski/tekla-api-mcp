DrawingItemExport Method

Exports the drawing item in the catalog to given file name. If path is not given drawing is exported to model folder. If filename is empty drawing name is used as filename.

using System;
using Tekla.Structures.Catalogs;

public class Example
{
       public void Example1()
       {
           DrawingItem item = new DrawingItem();

           if (item.Select())
           {
               string filename = System.IO.Path.GetTempPath() + item.Name;
               if (item.Export(ref filename))
               {
                   Console.WriteLine(item.Name + " exported successfully");
               }
           }
